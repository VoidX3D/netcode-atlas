JavaScript

The programming language that makes web pages interactive and dynamic

What is JavaScript?

JavaScript is a high-level, interpreted programming language that conforms to the ECMAScript specification. It is a language that is also characterized as dynamic, weakly typed, prototype-based and multi-paradigm.

JavaScript was created by Brendan Eich in 1995 while working at Netscape Communications. It was originally named Mocha, then LiveScript, and finally JavaScript.

JavaScript Fundamentals

Variables and Data Types

// Variable declaration
let message = "Hello";
const pi = 3.14159;
var count = 0;

// Data types
let string = "Text";
let number = 42;
let boolean = true;
let array = [1, 2, 3];
let object = { key: "value" };
let nullValue = null;
let undefinedValue;

Functions

// Function declaration
function greet(name) {
  return `Hello, ${name}!`;
}

// Arrow function (ES6+)
const multiply = (a, b) => a * b;

// Callback function
setTimeout(() => {
  console.log("Delayed message");
}, 1000);

Control Flow

// Conditional statements
if (age >= 18) {
  console.log("Adult");
} else if (age >= 13) {
  console.log("Teenager");
} else {
  console.log("Child");
}

// Loops
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// Switch statement
switch (day) {
  case "Monday":
    console.log("Start of work week");
    break;
  default:
    console.log("Other day");
}

Objects and Arrays

// Object literal
const person = {
  name: "Alice",
  age: 30,
  greet() {
    return `Hello, I'm ${this.name}`;
  }
};

// Array methods
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(n => n * 2);
const evens = numbers.filter(n => n % 2 === 0);
const sum = numbers.reduce((total, n) => total + n, 0);

DOM Manipulation

Selecting Elements

// Select single element
const header = document.querySelector('header');
const mainHeading = document.getElementById('main-heading');

// Select multiple elements
const paragraphs = document.querySelectorAll('p');
const listItems = document.getElementsByClassName('list-item');

Modifying Elements

// Change content
element.textContent = 'New text';
element.innerHTML = 'Bold text';

// Modify attributes
element.setAttribute('id', 'new-id');
element.classList.add('active');
element.classList.remove('hidden');

// Change styles
element.style.color = 'blue';
element.style.backgroundColor = '#f0f0f0';

Event Handling

Event Listeners

// Add event listener
button.addEventListener('click', function(event) {
  console.log('Button clicked!');
  event.preventDefault();
});

// Common events
document.addEventListener('DOMContentLoaded', init);
window.addEventListener('resize', handleResize);
form.addEventListener('submit', handleSubmit);
input.addEventListener('input', handleInput);

Event Delegation

// Instead of adding listeners to each item
listItems.forEach(item => {
  item.addEventListener('click', handleClick);
});

// Use event delegation on parent
list.addEventListener('click', function(event) {
  if (event.target.tagName === 'LI') {
    handleClick(event.target);
  }
});

Modern JavaScript (ES6+)

Template Literals

// Instead of concatenation
const greeting = 'Hello, ' + name + '!';

// Use template literals
const greeting = `Hello, ${name}!`;
const multiline = `
  This is a
  multiline string
`;

Destructuring

// Object destructuring
const { name, age, city } = person;
const { name: firstName, ...rest } = person;

// Array destructuring
const [first, second, ...others] = numbers;

Spread and Rest Operators

// Spread operator
const combined = [...array1, ...array2];
const clone = { ...original };

// Rest parameters
function sum(...numbers) {
  return numbers.reduce((total, n) => total + n, 0);
}

Promises and Async/Await

// Promises
fetch('/api/data')
  .then(response => response.json())
  .then(data => console.log(data))
  .catch(error => console.error(error));

// Async/await
async function getData() {
  try {
    const response = await fetch('/api/data');
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.error(error);
  }
}

JavaScript Modules

Exporting

// Named exports
export const pi = 3.14159;
export function calculateArea(radius) {
  return pi * radius * radius;
}

// Default export
export default class Calculator {
  // class implementation
}

Importing

// Import named exports
import { pi, calculateArea } from './math.js';

// Import default export
import Calculator from './calculator.js';

// Import everything
import * as math from './math.js';

Error Handling

Try/Catch

try {
  // Code that might throw an error
  const result = riskyOperation();
  console.log(result);
} catch (error) {
  // Handle the error
  console.error('An error occurred:', error.message);
  // Optionally re-throw the error
  throw error;
} finally {
  // Code that runs regardless of success/failure
  cleanup();
}

Custom Errors

class ValidationError extends Error {
  constructor(message) {
    super(message);
    this.name = 'ValidationError';
  }
}

function validateInput(input) {
  if (!input) {
    throw new ValidationError('Input cannot be empty');
  }
}
